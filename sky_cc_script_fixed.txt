-- ============================================================
-- EXECUTOR COMPATIBILITY SHIM
-- ============================================================
-- task library: patch only the slots that are missing.
-- NOTE: we do NOT blindly fall back to legacy wait() because that
-- global may also be nil in newer Roblox builds. Instead we build
-- a Heartbeat-based wait that works without any extra globals.
local _HB = game:GetService("RunService").Heartbeat
local function _safeWait(n)
    local t = 0
    while t < (n or 0) do t = t + _HB:Wait() end
end
if not task then task = {} end
if type(task.wait)  ~= "function" then task.wait  = _safeWait end
if type(task.spawn) ~= "function" then
    task.spawn = function(f, ...) local ok, err = pcall(coroutine.resume, coroutine.create(f), ...) end
end
if type(task.delay) ~= "function" then
    task.delay = function(n, f)
        task.spawn(function() task.wait(n); f() end)
    end
end
-- Guard file I/O (not all executors expose these)
if not readfile  then readfile  = function() return nil end end
if not writefile then writefile = function() end end

local function gradStroke(frame, thickness, glowing)
    local s = frame:FindFirstChildOfClass("UIStroke")
    if not s then
        s = Instance.new("UIStroke")
        s.Parent = frame
    end
    s.Color = Color3.fromRGB(255, 255, 255)
    s.Thickness = thickness or 1.5
    s.Transparency = 0.1
    s.ApplyStrokeMode = Enum.ApplyStrokeMode.Border

    local g = s:FindFirstChildOfClass("UIGradient")
    if not g then
        g = Instance.new("UIGradient")
        g.Color = SKY_GRAD
        g.Parent = s
    else
        g.Color = SKY_GRAD
    end
    table.insert(_shimmerGrads, g)

    if glowing then
        local glow = Instance.new("UIStroke")
        glow.ApplyStrokeMode = Enum.ApplyStrokeMode.Border
        glow.Color = SKY.ACCENT
        glow.Thickness = 3
        glow.Transparency = 0.75
        glow.Parent = frame
        table.insert(_glowStrokes, glow)
    end
end

-- Rotate shimmer gradients continuously (Heartbeat-based, no task.wait needed)
local _shimmerRot = 0
RunService.Heartbeat:Connect(function()
    _shimmerRot = (_shimmerRot + 1.6) % 360
    for i = #_shimmerGrads, 1, -1 do
        local g = _shimmerGrads[i]
        if g and g.Parent then
            g.Rotation = _shimmerRot
        else
            table.remove(_shimmerGrads, i)
        end
    end
end)

-- Breathe glow strokes (Heartbeat timer, no task.wait needed)
local _glowTimer, _glowExpand = 0, true
RunService.Heartbeat:Connect(function(dt)
    _glowTimer = _glowTimer + dt
    if _glowTimer < 2.4 then return end
    _glowTimer = 0
    _glowExpand = not _glowExpand
    local tw = _glowExpand and {Thickness = 5, Transparency = 0.5}

        else

        end
    end
end)

local function fileRead()
    local ok, raw = pcall(readfile, SAVE_FILE)
    if not ok or not raw or raw == "" then return nil end
    local dok, data = pcall(function() return HttpService:JSONDecode(raw) end)
    if dok and type(data) == "table" then return data end
    return nil
end
local function fileWrite(tbl)
    pcall(writefile, SAVE_FILE, HttpService:JSONEncode(tbl))
end

end

        if S[k] ~= nil then S[k] = v end
    end
end

local function saveSettings()
    _G.SkyCCSettings = S
    local t = {}
    for k in pairs(DEFAULT_SETTINGS) do t[k] = S[k] end
    fileWrite(t)
end

local _starRng = Random.new()
for _si = 1, 52 do
    local dot = Instance.new("Frame")
    local sz = _starRng:NextInteger(1, (_si % 8 == 0) and 3 or 2)
    dot.Size = UDim2.fromOffset(sz, sz)
    dot.Position = UDim2.new(_starRng:NextNumber(0.01, 0.99), 0, _starRng:NextNumber(0.01, 0.99), 0)
    local shade = _starRng:NextInteger(180, 255)
    dot.BackgroundColor3 = Color3.fromRGB(shade, shade, shade)
    dot.BackgroundTransparency = _starRng:NextNumber(0.2, 0.6)
    dot.BorderSizePixel = 0
    dot.ZIndex = 0
    dot.Parent = Main
    SkyCorner(dot, 999)

    local period = _starRng:NextNumber(0.9, 3.2)
    pcall(task.spawn, function()
        while dot.Parent do
            TweenService:Create(dot, TweenInfo.new(period, Enum.EasingStyle.Sine, Enum.EasingDirection.InOut),
                {BackgroundTransparency = _starRng:NextNumber(0.02, 0.35)}):Play()
            task.wait(period)
            if not dot.Parent then break end
            TweenService:Create(dot, TweenInfo.new(period, Enum.EasingStyle.Sine, Enum.EasingDirection.InOut),
                {BackgroundTransparency = _starRng:NextNumber(0.5, 0.85)}):Play()
            task.wait(period)
        end
    end)

    pcall(task.spawn, function()
        local dx = _starRng:NextNumber(-0.005, 0.005)
        local dy = _starRng:NextNumber(-0.003, 0.003)
        if math.abs(dx) < 0.001 then dx = 0.0018 end
        if math.abs(dy) < 0.001 then dy = 0.0012 end
        while dot.Parent do
            local cx = dot.Position.X.Scale
            local cy = dot.Position.Y.Scale
            local nx, ny = cx + dx, cy + dy
            if nx < 0.01 or nx > 0.99 then dx = -dx; nx = cx + dx end
            if ny < 0.01 or ny > 0.99 then dy = -dy; ny = cy + dy end
            dot.Position = UDim2.new(nx, 0, ny, 0)
            task.wait(0.05)
        end
    end)
end

    local dragging, dragStart, startPos = false
    TitleBar.InputBegan:Connect(function(input)
        if guiLocked then return end
        if input.UserInputType == Enum.UserInputType.MouseButton1
        or input.UserInputType == Enum.UserInputType.Touch then
            dragging = true; dragStart = input.Position; startPos = Main.Position
            input.Changed:Connect(function()
                if input.UserInputState == Enum.UserInputState.End then dragging = false end
            end)
        end
    end)
    UserInputService.InputChanged:Connect(function(input)
        if dragging and not guiLocked and
        (input.UserInputType == Enum.UserInputType.MouseMovement
        or input.UserInputType == Enum.UserInputType.Touch) then
            local d = input.Position - dragStart
            Main.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + d.X, startPos.Y.Scale, startPos.Y.Offset + d.Y)
        end
    end)
    UserInputService.InputEnded:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1
        or input.UserInputType == Enum.UserInputType.Touch then dragging = false end
    end)
end

local function findToolByNames(nameList, blockedList)
    local char = player.Character
    local bp = player:FindFirstChild("Backpack")
    for _, src in ipairs({char, bp}) do
        if src then
            for _, item in ipairs(src:GetChildren()) do
                if item:IsA("Tool") then
                    local low = item.Name:lower()
                    local blocked = false
                    if blockedList then
                        for _, b in ipairs(blockedList) do
                            if low:find(b, 1, true) then blocked = true; break end
                        end
                    end
                    if not blocked then
                        for _, n in ipairs(nameList) do
                            if low:find(n:lower(), 1, true) then return item, src end
                        end
                    end
                end
            end
        end
    end
    return nil, nil
end

local function equipTool(tool)
    local char = player.Character
    if not char then return false end
    local hum = char:FindFirstChildOfClass("Humanoid")
    if not hum then return false end
    if tool.Parent == char then return true end
    pcall(function()
        tool.Parent = char
        hum:EquipTool(tool)
    end)
    local t = tick()
    while tick() - t < 0.3 do
        if tool.Parent == char then return true end
        task.wait()
    end
    return tool.Parent == char
end

local function alignCamera(overrideLook)
    local camera = workspace.CurrentCamera
    if not camera then return end
    local char = player.Character
    local root = char and char:FindFirstChild("HumanoidRootPart")
    if not root then return end
    local fwd = overrideLook or root.CFrame.LookVector
    fwd = Vector3.new(fwd.X, 0, fwd.Z)
    if fwd.Magnitude < 0.01 then fwd = root.CFrame.LookVector end
    fwd = fwd.Unit
    local camPos = root.Position - fwd * 3 + Vector3.new(0, -2, 0)
    local lookAt = root.Position + fwd * 5 + Vector3.new(0, 3.5, 0)
    camera.CFrame = CFrame.lookAt(camPos, lookAt)
end

local function applyFPSBooster(on)
    pcall(function()
        if on then
            settings().Rendering.QualityLevel = Enum.QualityLevel.Level01
            Lighting.GlobalShadows = false; Lighting.FogEnd = 9e8; Lighting.Brightness = 1
            Lighting.EnvironmentDiffuseScale = 0; Lighting.EnvironmentSpecularScale = 0
            for _, e in pairs(Lighting:GetChildren()) do
                if e:IsA("BlurEffect") or e:IsA("SunRaysEffect")
                or e:IsA("ColorCorrectionEffect") or e:IsA("BloomEffect")
                or e:IsA("DepthOfFieldEffect") then pcall(function() e.Enabled = false end) end
            end
            for _, obj in ipairs(workspace:GetDescendants()) do
                pcall(function()
                    if obj:IsA("BasePart") or obj:IsA("MeshPart") then
                        obj.Material = Enum.Material.Plastic; obj.Reflectance = 0; obj.CastShadow = false
                    end
                    if obj:IsA("ParticleEmitter") or obj:IsA("Trail") or obj:IsA("Beam")
                    or obj:IsA("Fire") or obj:IsA("Smoke") or obj:IsA("Sparkles") then
                        obj.Enabled = false
                    end
                    if obj:IsA("Decal") or obj:IsA("Texture") then obj.Transparency = 1 end
                end)
            end
        else
            settings().Rendering.QualityLevel = Enum.QualityLevel.Automatic
            Lighting.GlobalShadows = true; Lighting.FogEnd = 100000
            Lighting.EnvironmentDiffuseScale = 1; Lighting.EnvironmentSpecularScale = 1
        end
    end)
end

local function applyAntiLag()
    local removed = 0
    for _, obj in ipairs(workspace:GetDescendants()) do

            pcall(function() obj:Destroy() end); removed = removed + 1
        end
    end
    return removed
end

local function updateBaseESPTick()
    if not S.baseESP then return end
    local plotsFolder = workspace:FindFirstChild("Plots"); if not plotsFolder then return end
    for _, plot in ipairs(plotsFolder:GetChildren()) do
        local plotName = plot.Name
        local timeLabel, mainPart = nil, nil
        local purchases = plot:FindFirstChild("Purchases")
        local plotBlock = purchases and purchases:FindFirstChild("PlotBlock")
        local mp = plotBlock and plotBlock:FindFirstChild("Main")
        if mp then
            for _, child in ipairs(mp:GetChildren()) do
                if child:IsA("BillboardGui") then
                    local rt = child:FindFirstChild("RemainingTime")
                    if rt and rt:IsA("TextLabel") then timeLabel = rt; mainPart = mp; break end
                end
            end
        end
        if not timeLabel then
            for _, desc in ipairs(plot:GetDescendants()) do
                if desc.Name == "RemainingTime" and desc:IsA("TextLabel") then
                    timeLabel = desc
                    local p = desc.Parent
                    while p and p ~= plot do
                        if p:IsA("BasePart") then mainPart = p; break end
                        p = p.Parent
                    end
                    if not mainPart and mp then mainPart = mp end
                    break
                end
            end
        end
        if timeLabel and mainPart then
            local existing = baseEspInstances[plotName]
            local bb
            if existing and existing.Parent and existing.Parent == mainPart then
                bb = existing
            else
                bb = createBaseESP(plotName, mainPart)
            end
            local bg = bb:FindFirstChild("Frame")
            local lbl = bg and bg:FindFirstChild("TimerText")
            if lbl then lbl.Text = timeLabel.Text end
        elseif baseEspInstances[plotName] then
            pcall(function() baseEspInstances[plotName]:Destroy() end)
            baseEspInstances[plotName] = nil
        end
    end
end

local function parsePetValue(text)
    if not text then return 0 end
    text = tostring(text):gsub("%s", ""):gsub("/s", "")
    local numStr, suffix = text:match("([%d%.]+)([KkMmBbTt]?)")
    if not numStr then return 0 end
    local num = tonumber(numStr) or 0
    local mult = ({K=1e3,M=1e6,B=1e9,T=1e12})[(suffix or ""):upper()] or 1
    return num * mult
end

local function scanBestPet()
    local debris = workspace:FindFirstChild("Debris"); if not debris then return end
    local best = {value=-1, part=nil, name="", raw=""}
    for _, item in ipairs(debris:GetChildren()) do
        if item.Name == "FastOverheadTemplate" then
            local sg = item:FindFirstChildOfClass("SurfaceGui")
            if sg and sg.Adornee then
                local gl = sg:FindFirstChild("Generation", true)
                if gl and gl:IsA("TextLabel") then
                    local v = parsePetValue(gl.Text)
                    if v > best.value then
                        local nl = sg:FindFirstChild("DisplayName", true)
                        best = {value=v, part=sg.Adornee, name=nl and nl.Text or "?", raw=gl.Text}
                    end
                end
            end
        end
    end
    local bb = getBestPetBB()
    if best.part and bestPetActive then
        bb.Adornee = best.part; bb.Enabled = true
        local f = bb:FindFirstChild("Frame")
        if f then
            local nl = f:FindFirstChild("PetName"); local vl = f:FindFirstChild("PetValue")
            if nl then nl.Text = best.name end; if vl then vl.Text = best.raw end
        end
    else
        bb.Enabled = false; bb.Adornee = nil
    end
end

local antiRagCharConn = nil

local function applyAntiRagdollToChar(char)
    if not char then return end
    local hum = char:FindFirstChildOfClass("Humanoid"); if not hum then return end
    pcall(function() hum:SetStateEnabled(Enum.HumanoidStateType.Ragdoll, false) end)
    pcall(function() hum:SetStateEnabled(Enum.HumanoidStateType.FallingDown, false) end)
    pcall(function() hum:SetStateEnabled(Enum.HumanoidStateType.Physics, false) end)
end

local function startAntiRagdoll()
    applyAntiRagdollToChar(player.Character)
    if antiRagCharConn then antiRagCharConn:Disconnect() end
    antiRagCharConn = player.CharacterAdded:Connect(function(char)
        if S.antiRagdoll then task.wait(0.1); applyAntiRagdollToChar(char) end
    end)
    task.spawn(function()
        while S.antiRagdoll do
            task.wait(0.25)
            local char = player.Character
            local hum = char and char:FindFirstChildOfClass("Humanoid")
            if char and hum then
                local root = char:FindFirstChild("HumanoidRootPart")
                local st = hum:GetState()
                if st == Enum.HumanoidStateType.Ragdoll
                or st == Enum.HumanoidStateType.FallingDown
                or st == Enum.HumanoidStateType.Physics then
                    pcall(function() hum:SetStateEnabled(Enum.HumanoidStateType.Ragdoll, false) end)
                    pcall(function() hum:SetStateEnabled(Enum.HumanoidStateType.FallingDown, false) end)
                    pcall(function() hum:SetStateEnabled(Enum.HumanoidStateType.Physics, false) end)
                    hum:ChangeState(Enum.HumanoidStateType.GettingUp)
                    if root then
                        pcall(function()
                            root.AssemblyLinearVelocity = Vector3.new(
                                root.AssemblyLinearVelocity.X * 0.1,
                                math.max(root.AssemblyLinearVelocity.Y, 0),
                                root.AssemblyLinearVelocity.Z * 0.1)
                            root.AssemblyAngularVelocity = Vector3.zero
                        end)
                    end
                    if workspace.CurrentCamera then workspace.CurrentCamera.CameraSubject = hum end
                end
            end
        end
    end)
end

local function stopAntiRagdoll()
    if antiRagCharConn then antiRagCharConn:Disconnect(); antiRagCharConn = nil end
    local char = player.Character; if not char then return end
    local hum = char:FindFirstChildOfClass("Humanoid"); if not hum then return end

    pcall(function() hum:SetStateEnabled(Enum.HumanoidStateType.FallingDown, true) end)
    pcall(function() hum:SetStateEnabled(Enum.HumanoidStateType.Physics, true) end)
end

-- FLASH TP LOGIC (unchanged)

local activeTriggers = {}
local triggerCount   = 0

local function newTrigId()
    triggerCount = triggerCount + 1
    return triggerCount
end

local function flashTeleport(prompt, targetPos)
    if not S.flashEnabled then return end
    local char = player.Character; if not char then return end
    local root = char:FindFirstChild("HumanoidRootPart"); if not root then return end
    local trig, tl = findToolByNames({"trigger","fist","punch","syringe","flash"}, {"giant","mega","colossal"})
    if not trig then return end
    local tId = newTrigId(); activeTriggers[tId] = true
    local originalPos = root.Position
    root.CFrame = CFrame.new(targetPos)
    task.wait(0.05)
    local held = false
    local holdConn
    holdConn = RunService.Heartbeat:Connect(function()
        if not activeTriggers[tId] then if holdConn then holdConn:Disconnect() end; return end
        pcall(function()
            if prompt and prompt.Parent then
                ProximityPromptService:HoldBegin(prompt)
                held = true
            end
        end)
    end)
    local threshold = S.sliderValue or 0.91
    local pct = 0
    StealBarBG.Visible = true; StealBarTopLabel.Visible = true
    local fillConn
    fillConn = RunService.Heartbeat:Connect(function(dt)
        if not activeTriggers[tId] then
            if fillConn then fillConn:Disconnect() end
            StealBarBG.Visible = false; StealBarTopLabel.Visible = false
            StealBarFill.Size = UDim2.new(0, 0, 1, 0)
            BarPercentLabel.Text = ""; return
        end
        pct = math.min(pct + dt * (1 / (prompt and prompt.HoldDuration or 1)), 1)
        StealBarFill.Size = UDim2.new(pct, 0, 1, 0)
        BarPercentLabel.Text = math.floor(pct * 100) .. "%"
        if pct >= threshold then
            activeTriggers[tId] = nil
            if holdConn then holdConn:Disconnect() end
            if fillConn then fillConn:Disconnect() end
            pcall(function()
                if prompt and prompt.Parent then ProximityPromptService:HoldEnd(prompt) end
            end)
            task.wait(0.1)
            root.CFrame = CFrame.new(originalPos)
            StealBarBG.Visible = false; StealBarTopLabel.Visible = false
            StealBarFill.Size = UDim2.new(0, 0, 1, 0); BarPercentLabel.Text = ""
        end
    end)
    task.delay(10, function()
        if activeTriggers[tId] then
            activeTriggers[tId] = nil
            if holdConn then holdConn:Disconnect() end
            if fillConn then fillConn:Disconnect() end
            pcall(function()
                if prompt and prompt.Parent then ProximityPromptService:HoldEnd(prompt) end
            end)
            root.CFrame = CFrame.new(originalPos)
            StealBarBG.Visible = false; StealBarTopLabel.Visible = false
            StealBarFill.Size = UDim2.new(0, 0, 1, 0); BarPercentLabel.Text = ""
        end
    end)
end

-- Proximity prompt listener
ProximityPromptService.PromptButtonHoldBegan:Connect(function(prompt, plr)
    if plr ~= player then return end
    if not S.flashEnabled then return end
    local targ = prompt.Parent
    local targetPos = (targ and targ:IsA("BasePart")) and targ.Position
        or (targ and targ:IsA("Model") and targ.PrimaryPart and targ.PrimaryPart.Position)
    if not targetPos then return end
    task.spawn(function() flashTeleport(prompt, targetPos) end)
end)

local giantConn = nil
local function startGiant()
    if giantConn then giantConn:Disconnect() end
    giantConn = RunService.Heartbeat:Connect(function()
        if not S.giantEnabled then if giantConn then giantConn:Disconnect(); giantConn = nil end; return end
        local char = player.Character; if not char then return end
        local bp = player:FindFirstChildOfClass("Backpack")
        for _, src in ipairs({char, bp}) do
            if src then
                for _, tool in ipairs(src:GetChildren()) do
                    if tool:IsA("Tool") and tool.Name:lower():find("giant") then
                        local hum = char:FindFirstChildOfClass("Humanoid")
                        if hum then
                            pcall(function() tool.Parent = char; hum:EquipTool(tool) end)
                            pcall(function()
                                local activate = tool:FindFirstChild("RemoteEvent") or tool:FindFirstChild("ClickDetector")
                                if activate and activate:IsA("RemoteEvent") then activate:FireServer() end
                            end)
                        end
                        break
                    end
                end
            end
        end
    end)
end

local function brp_parseMoney(text)
    text = string.lower(text or ""); local n = tonumber(text:match("[%d%.]+")) or 0
    if text:find("k") then n = n*1e3 elseif text:find("m") then n = n*1e6
    elseif text:find("b") then n = n*1e9 elseif text:find("t") then n = n*1e12
    elseif text:find("qa") then n = n*1e15 end
    return n
end

local function brp_readDG(part)
    local dn, gen
    for _, c in ipairs(part:GetChildren()) do
        if c:IsA("BillboardGui") or c:IsA("SurfaceGui") then
            for _, lbl in ipairs(c:GetDescendants()) do
                if lbl:IsA("TextLabel") then
                    if lbl.Name == "DisplayName" then dn = lbl.Text
                    elseif lbl.Name == "Generation" then gen = lbl.Text end
                end
            end
        end
    end
    return dn, gen
end

local function brp_findBaseBB()
    local vis = {}
    for _, o in ipairs(workspace:GetDescendants()) do
        if o:IsA("BillboardGui") and o.Enabled then
            for _, c in ipairs(o:GetDescendants()) do
                if c:IsA("TextLabel") and c.Visible and string.upper(c.Text or "") == "YOUR BASE" and c.TextTransparency < 1 then
                    table.insert(vis, o); break
                end
            end
        end
    end
    if #vis > 1 and player.Character then

        if root then
            local cl, cd
            for _, bb in ipairs(vis) do
                local pos

                elseif bb.Parent and bb.Parent:IsA("Model") then

                end
                if pos then

                    if not cl or d < cd then cd = d; cl = bb end
                end
            end
            if cl then return cl end
        end

local function brp_findPodiums(plot)

    for _, d in ipairs(plot:GetDescendants()) do
        if (d:IsA("Folder") or d:IsA("Model")) and d.Name:lower():find("podium") and not tonumber(d.Name) then
            for _, c in ipairs(d:GetChildren()) do
                local n = brp_podiumNum(c)
                if n and not seen[c] then
                    seen[c] = true
                    local pos = brp_podiumPos(c)
                    if pos then table.insert(r, {podium=c, number=n, position=pos}) end
                end
            end
        end
    end
    if #r == 0 then
        local nb = {}
        for _, d in ipairs(plot:GetDescendants()) do
            if (d:IsA("Model") or d:IsA("Folder") or d:IsA("BasePart")) and tonumber(d.Name) then
                local n = tonumber(d.Name)
                if n >= 1 and n <= 50 then table.insert(nb, {obj=d, num=n}) end
            end
        end
        for _, it in ipairs(nb) do
            if not seen[it.obj] then
                seen[it.obj] = true
                local pos = brp_podiumPos(it.obj)
                if pos then table.insert(r, {podium=it.obj, number=it.num, position=pos}) end
            end
        end
    end
    if #r == 0 then
        for _, d in ipairs(plot:GetDescendants()) do
            if d:IsA("BasePart") and not seen[d] then
                local n = brp_podiumNum(d)
                if n then
                    seen[d] = true
                    local pos = brp_podiumPos(d)
                    if pos then table.insert(r, {podium=d, number=n, position=pos}) end
                end
            end
        end
    end
    return r
end

local function brp_ownerName(plot)
    local ps = plot:FindFirstChild("PlotSign", true)
    if ps then
        for _, d in ipairs(ps:GetDescendants()) do
            if d:IsA("TextLabel") and d.Text and d.Text ~= "" then
                local t = d.Text
                if t:upper() == "YOUR BASE" then return "YOU" end
                if t:lower():find("empty") then return "Empty" end
                local m = t:match("[Bb]ase [Oo]f%s+(.+)")
                if m then return m end
                if #t > 0 and #t < 30 then return t end
            end
        end
    end
    return plot.Name
end

local function brp_isMyPlot(plot, myPlot)
    if myPlot and plot == myPlot then return true end
    for _, d in ipairs(plot:GetDescendants()) do
        if d:IsA("BillboardGui") and d.Enabled then
            for _, c in ipairs(d:GetDescendants()) do
                if c:IsA("TextLabel") and c.Visible and string.upper(c.Text or "") == "YOUR BASE" and c.TextTransparency < 1 then
                    return true
                end
            end
        end
    end
    return false
end

local function brp_bounds(plot)
    local minX, maxX, minZ, maxZ = math.huge, -math.huge, math.huge, -math.huge
    local found = false
    for _, d in ipairs(plot:GetDescendants()) do
        if d:IsA("BasePart") then
            local p = d.Position
            if p.X < minX then minX = p.X end; if p.X > maxX then maxX = p.X end
            if p.Z < minZ then minZ = p.Z end; if p.Z > maxZ then maxZ = p.Z end
            found = true
        end
    end
    if not found then return nil end
    return {minX=minX, maxX=maxX, minZ=minZ, maxZ=maxZ}
end

local function brp_animalsInPlot(animals, plot, bounds)
    local r = {}; local pad = 4
    for _, a in ipairs(animals) do
        local inside = a.part:IsDescendantOf(plot)
        if not inside and bounds then
            local p = a.position
            inside = p.X >= bounds.minX - pad and p.X <= bounds.maxX + pad
                and p.Z >= bounds.minZ - pad and p.Z <= bounds.maxZ + pad
        end
        if inside then table.insert(r, a) end
    end
    return r
end

local function brp_match(podiums, animals)
    local cand = {}
    for pi, p in ipairs(podiums) do
        for ai, a in ipairs(animals) do
            local diff = a.position - p.position
            local hd = math.sqrt(diff.X*diff.X + diff.Z*diff.Z)
            local yd = math.abs(diff.Y)
            if hd < BRP_MATCH_DIST and yd < 15 then
                table.insert(cand, {pi=pi, ai=ai, score=math.sqrt(hd*hd + yd*yd*25)})
            end
        end
    end
    table.sort(cand, function(a, b) return a.score < b.score end)
    local p2a, uA, uP = {}, {}, {}
    for _, c in ipairs(cand) do
        if not uP[c.pi] and not uA[c.ai] then
            p2a[c.pi] = c.ai; uP[c.pi] = true; uA[c.ai] = true
        end
    end
    local entries = {}
    for pi, p in ipairs(podiums) do
        local e = {number=p.number, podium=p.podium, podiumPosition=p.position}
        local m = p2a[pi]
        if m then
            local a = animals[m]
            e.name = a.displayName; e.generation = a.generation; e.value = a.value
            e.animalPart = a.part; e.animalPosition = a.position; e.empty = false
        else e.empty = true; e.value = 0 end
        table.insert(entries, e)
    end
    return entries
end

local function brp_lookCF(podium, pos)
    local part
    if podium:IsA("BasePart") then part = podium
    elseif podium:IsA("Model") and podium.PrimaryPart then part = podium.PrimaryPart
    else part = podium:FindFirstChildWhichIsA("BasePart", true) end
    if part then return part.CFrame end
    return CFrame.new(pos)
end

local function brp_topY(podium, fallback)
    if podium:IsA("BasePart") then return podium.Position.Y + podium.Size.Y/2 end
    if podium:IsA("Model") then
        local ok, cf, sz = pcall(function() return podium:GetBoundingBox() end)
        if ok and cf and sz then return cf.Position.Y + sz.Y/2 end
        if podium.PrimaryPart then return podium.PrimaryPart.Position.Y + podium.PrimaryPart.Size.Y/2 end
        local part = podium:FindFirstChildWhichIsA("BasePart", true)
        if part then return part.Position.Y + part.Size.Y/2 end
    end
    return fallback.Y
end

local function brp_teleport(entry)
    local char = player.Character; if not char then return end
    local hrp = char:FindFirstChild("HumanoidRootPart")
    local hum = char:FindFirstChildOfClass("Humanoid")
    if not hrp or not hum then return nil end
    brp_equipCarpet(char)
    local pos = entry.podiumPosition; if not pos then return nil end
    local podCF = brp_lookCF(entry.podium, pos)
    local lookDir = -podCF.LookVector
    lookDir = Vector3.new(lookDir.X, 0, lookDir.Z)
    if lookDir.Magnitude < 0.1 then lookDir = Vector3.new(0, 0, 1) end
    lookDir = lookDir.Unit
    local topY = brp_topY(entry.podium, pos)
    local standX = entry.animalPosition and entry.animalPosition.X or pos.X
    local standZ = entry.animalPosition and entry.animalPosition.Z or pos.Z
    local standY = entry.animalPosition and entry.animalPosition.Y or (topY + 2)
    local standPos = Vector3.new(standX, standY, standZ)
    local walkTarget = Vector3.new(standPos.X, standPos.Y, standPos.Z)
    local startTime = tick()
    hum:MoveTo(walkTarget)
    while tick() - startTime < 6 do
        if not hrp.Parent then break end
        local diff = walkTarget - hrp.Position
        local fd = Vector3.new(diff.X, 0, diff.Z)
        if fd.Magnitude < 3 then break end
        local md = hum.MoveDirection
        local dir = md.Magnitude > 0 and Vector3.new(md.X, 0, md.Z).Unit or fd.Unit
        hrp.Velocity = Vector3.new(dir.X * BRP_WALK_SPEED, hrp.Velocity.Y, dir.Z * BRP_WALK_SPEED)
        task.wait()
    end
    hum:Move(Vector3.zero, false)
    hum:MoveTo(hrp.Position)
    task.wait(0.2)
    hrp.CFrame = CFrame.lookAt(standPos, standPos + lookDir)
    hrp.Velocity = Vector3.zero
    task.spawn(function()
        local snapCF = CFrame.lookAt(standPos, standPos + lookDir)
        for _ = 1, 5 do
            if not hrp.Parent then break end
            hrp.CFrame = snapCF; hrp.Velocity = Vector3.zero; task.wait()
        end
    end)
    local camera = workspace.CurrentCamera
    if camera then
        task.spawn(function()
            local ot = camera.CameraType
            camera.CameraType = Enum.CameraType.Scriptable
            local camPos = standPos + (-lookDir * BRP_CAM_BACK) + Vector3.new(0, BRP_CAM_HEIGHT, 0)
            local lt = standPos + lookDir * 4 + Vector3.new(0, BRP_CAM_HEIGHT + BRP_CAM_PITCH, 0)
            local tCF = CFrame.lookAt(camPos, lt)
            for _ = 1, 12 do
                if not hrp.Parent then break end
                camera.CFrame = tCF; task.wait()
            end
            camera.CameraType = ot
        end)
    end
    return lookDir
end

local function brp_getAnimalModel(data)
    if data and data.name then
        local ok, v = pcall(function()
            local rs = game:GetService("ReplicatedStorage")
            local models = rs:FindFirstChild("Models")
            local animals = models and models:FindFirstChild("Animals")
            return animals and animals:FindFirstChild(data.name) or nil
        end)
        if ok and v then return v end
    end
    if data and data.animalPart and data.animalPart.Parent then
        local m = data.animalPart:FindFirstAncestorOfClass("Model")
        if m then return m end
    end
    return nil
end

local function brp_makeViewport(parent, data)
    local vp = Instance.new("ViewportFrame")
    vp.Size = UDim2.new(0, 58, 0, 58); vp.Position = UDim2.new(0, 6, 0.5, -29)
    vp.BackgroundColor3 = SKY.SURF; vp.BackgroundTransparency = 0.1
    vp.BorderSizePixel = 0
    vp.LightDirection = Vector3.new(-1, -2, -1)
    vp.LightColor = Color3.fromRGB(220, 220, 255)
    vp.Ambient = Color3.fromRGB(180, 180, 180)
    vp.ZIndex = 12; vp.Parent = parent
    SkyCorner(vp, 5)
    do local s = Instance.new("UIStroke"); s.Color = T.Border; s.Thickness = 1; s.Parent = vp end
    local model = brp_getAnimalModel(data)
    if not model then return vp end
    local wm = Instance.new("WorldModel"); wm.Parent = vp
    local cam = Instance.new("Camera"); cam.Parent = vp; vp.CurrentCamera = cam
    task.spawn(function()
        local ok, clone = pcall(function() return model:Clone() end)
        if not ok or not clone then return end
        for _, d in ipairs(clone:GetDescendants()) do
            if d:IsA("BasePart") then d.Anchored = true; d.CanCollide = false
            elseif d:IsA("ParticleEmitter") or d:IsA("Trail") or d:IsA("Smoke") or d:IsA("Fire") or d:IsA("Sparkles") then
                pcall(function() d.Enabled = false end)
            elseif d:IsA("Script") or d:IsA("LocalScript") then d:Destroy() end
        end
        clone.Parent = wm
        local cf, size
        if clone:IsA("Model") then cf, size = clone:GetBoundingBox()
        else cf = clone.CFrame; size = clone.Size end
        local maxDim = math.max(size.X, size.Y, size.Z)
        if maxDim <= 0 then return end
        local dist = maxDim * 1.25
        cam.FieldOfView = 55
        cam.CFrame = CFrame.new(cf.Position + Vector3.new(dist*0.65, maxDim*0.25, dist*0.65), cf.Position)
    end)
    return vp
end

local function brp_populateRows(plot)
    for _, c in ipairs(BRPScroll:GetChildren()) do
        if c:IsA("Frame") then c:Destroy() end
    end
    if not plot then
        BRPStatus.Text = "no plot found"
        BRPStatus.TextColor3 = RED
        return
    end
    BRPStatus.Text = "scanning..."
    BRPStatus.TextColor3 = SKY.DIM
    task.spawn(function()
        local animals  = brp_findAllAnimals()
        local podiums  = brp_findPodiums(plot)
        local bounds   = brp_bounds(plot)
        local plotAni  = brp_animalsInPlot(animals, plot, bounds)
        local entries  = brp_match(podiums, plotAni)
        table.sort(entries, function(a, b) return a.number < b.number end)
        if #entries == 0 then
            BRPStatus.Text = "no podiums found"; BRPStatus.TextColor3 = RED; return
        end
        for _, e in ipairs(entries) do brp_makeRow(e) end
        local filled = 0
        for _, e in ipairs(entries) do if not e.empty then filled = filled + 1 end end
        BRPStatus.Text = filled .. "/" .. #entries .. " slots * " .. brp_ownerName(plot)
        BRPStatus.TextColor3 = T.Good
    end)
end

-- Plot picker rows (shown first so user picks which plot)
local function brp_showPlotPicker()
    for _, c in ipairs(BRPScroll:GetChildren()) do
        if c:IsA("Frame") then c:Destroy() end
    end
    BRPStatus.Text = "pick a plot"
    BRPStatus.TextColor3 = SKY.DIM
    local plots = brp_findAllPlots()
    local myPlot = brp_getMyPlot()
    if #plots == 0 then
        BRPStatus.Text = "no plots found"; BRPStatus.TextColor3 = RED; return
    end
    for _, plot in ipairs(plots) do
        local isMe = brp_isMyPlot(plot, myPlot)
        local owner = brp_ownerName(plot)
        local pRow = Instance.new("TextButton")
        pRow.Size = UDim2.new(1, 0, 0, 36)
        pRow.BackgroundColor3 = isMe and Color3.fromRGB(10, 30, 10) or SKY.SURF
        pRow.BackgroundTransparency = 0.1
        pRow.BorderSizePixel = 0
        pRow.Text = ""
        pRow.AutoButtonColor = false
        pRow.ZIndex = 12
        pRow.Parent = BRPScroll
        SkyCorner(pRow, 8)
        do local s = Instance.new("UIStroke"); s.Color = isMe and GREEN or Color3.fromRGB(55,55,55); s.Thickness = 1; s.Transparency = 0.2; s.Parent = pRow end

        local pLbl = Instance.new("TextLabel")
        pLbl.Size = UDim2.new(1, -60, 1, 0)
        pLbl.Position = UDim2.new(0, 12, 0, 0)
        pLbl.BackgroundTransparency = 1
        pLbl.Text = (isMe and "* " or "") .. owner
        pLbl.TextColor3 = isMe and GREEN or SKY.TEXT
        pLbl.Font = Enum.Font.GothamBold
        pLbl.TextSize = 12
        pLbl.TextXAlignment = Enum.TextXAlignment.Left
        pLbl.ZIndex = 13
        pLbl.Parent = pRow

        local pNumLbl = Instance.new("TextLabel")
        pNumLbl.Size = UDim2.new(0, 50, 1, 0)
        pNumLbl.Position = UDim2.new(1, -54, 0, 0)
        pNumLbl.BackgroundTransparency = 1
        pNumLbl.Text = plot.Name
        pNumLbl.TextColor3 = SKY.DIM
        pNumLbl.Font = Enum.Font.Gotham
        pNumLbl.TextSize = 10
        pNumLbl.TextXAlignment = Enum.TextXAlignment.Right
        pNumLbl.ZIndex = 13
        pNumLbl.Parent = pRow

        pRow.MouseButton1Click:Connect(function()
            currentPlot = plot
            brp_populateRows(plot)
        end)
        pRow.MouseEnter:Connect(function()
            TweenService:Create(pRow, TweenInfo.new(0.08), {BackgroundTransparency = 0}):Play()
        end)
        pRow.MouseLeave:Connect(function()
            TweenService:Create(pRow, TweenInfo.new(0.08), {BackgroundTransparency = 0.1}):Play()
        end)
    end
end

-- Refresh button
BRPRefresh.MouseButton1Click:Connect(function()
    BRPRefresh.Text = ">  SCANNING"
    BRPRefresh.TextColor3 = T.Accent
    TweenService:Create(BRPRefresh, TweenInfo.new(0.1), {BackgroundColor3 = SKY.SURF3}):Play()
    task.spawn(function()
        if currentPlot then
            brp_populateRows(currentPlot)
        else
            brp_showPlotPicker()
        end
        task.wait(0.4)
        BRPRefresh.Text = ">  REFRESH"
        BRPRefresh.TextColor3 = SKY.TEXT
        TweenService:Create(BRPRefresh, TweenInfo.new(0.1), {BackgroundColor3 = SKY.SURF2}):Play()
    end)
end)

-- Auto toggle
BRPAutoToggle.MouseButton1Click:Connect(function()
    brpAutoEnabled = not brpAutoEnabled
    if brpAutoEnabled then
        BRPAutoToggle.Text = "AUTO: ON"
        BRPAutoToggle.TextColor3 = GREEN
        TweenService:Create(BRPAutoToggle, TweenInfo.new(0.1), {BackgroundColor3 = Color3.fromRGB(10,30,10)}):Play()
        if brpAutoConn then brpAutoConn:Disconnect() end
        brpAutoConn = RunService.Heartbeat:Connect(function()
            if not brpAutoEnabled then
                brpAutoConn:Disconnect(); brpAutoConn = nil; return
            end
            if not currentPlot then return end
            local animals = brp_findAllAnimals()
            local podiums = brp_findPodiums(currentPlot)
            local bounds  = brp_bounds(currentPlot)
            local plotAni = brp_animalsInPlot(animals, currentPlot, bounds)
            local entries = brp_match(podiums, plotAni)
            for _, e in ipairs(entries) do
                if not e.empty and e.value > 0 then
                    task.spawn(function() brp_teleport(e) end)
                    task.wait(6)
                end
            end
        end)
    else
        BRPAutoToggle.Text = "AUTO: OFF"
        BRPAutoToggle.TextColor3 = SKY.DIM
        TweenService:Create(BRPAutoToggle, TweenInfo.new(0.1), {BackgroundColor3 = SKY.SURF2}):Play()
        if brpAutoConn then brpAutoConn:Disconnect(); brpAutoConn = nil end
    end
end)

-- Close
BRPClose.MouseButton1Click:Connect(function()
    S.brainrotPanelOpen = false; saveSettings()
    TweenService:Create(BrainrotPanel, TweenInfo.new(0.2, Enum.EasingStyle.Back, Enum.EasingDirection.In),
        {Size = UDim2.new(0, 320, 0, 0)}):Play()
    task.delay(0.2, function() BrainrotPanel.Visible = false; BrainrotPanel.Size = UDim2.new(0, 320, 0, 440) end)
end)

BrainrotToggleBtn.BackgroundColor3 = SKY.SURF2
gradStroke(BrainrotToggleBtn, 1)

BrainrotToggleBtn.MouseButton1Click:Connect(function()
    S.brainrotPanelOpen = not S.brainrotPanelOpen; saveSettings()
    if S.brainrotPanelOpen then
        BrainrotPanel.Visible = true
        BrainrotPanel.Size = UDim2.new(0, 320, 0, 0)
        TweenService:Create(BrainrotPanel, TweenInfo.new(0.28, Enum.EasingStyle.Back, Enum.EasingDirection.Out),
            {Size = UDim2.new(0, 320, 0, 440)}):Play()
        BrainrotToggleLbl.Text = "BRAINROT PANEL: ON"
        BrainrotToggleLbl.TextColor3 = GREEN
        task.spawn(brp_showPlotPicker)
    else
        BrainrotToggleLbl.Text = "BRAINROT PANEL"
        BrainrotToggleLbl.TextColor3 = SKY.TEXT
        TweenService:Create(BrainrotPanel, TweenInfo.new(0.2, Enum.EasingStyle.Back, Enum.EasingDirection.In),
            {Size = UDim2.new(0, 320, 0, 0)}):Play()
        task.delay(0.2, function() BrainrotPanel.Visible = false; BrainrotPanel.Size = UDim2.new(0, 320, 0, 440) end)
    end
end)

    BrainrotToggleLbl.Text = "BRAINROT PANEL: ON"
    BrainrotToggleLbl.TextColor3 = GREEN
    task.spawn(brp_showPlotPicker)
end

end)()

end, function(e)

end)
